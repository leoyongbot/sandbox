105 TomoPose trackingObject(const TomoPose& current_eef_pose, const TomoPose& current_tip_pose, const TomoPose& tomo_object_pose, int count,
106 int side)
107 {
108 TomoPose offset_pose_1 = current_tip_pose;
109 offset_pose_1.offsetPos(tomo_object_pose.pos);
110
111 auto m0 = current_eef_pose.getRotationMat();
112 double current_roll, current_pitch, current_yaw;
113 m0.getRPY(current_roll, current_pitch, current_yaw);
114 double x_local = current_tip_pose.pos[0] - current_eef_pose.pos[0];
115 double y_local = current_tip_pose.pos[1] - current_eef_pose.pos[1];
116 double x_offset_rotate, y_offset_rotate;
117
118 if(side == 0 && count < 11) {
119 current_yaw = current_yaw + count * (0.15708);
120 x_offset_rotate = x_local * cos(count * (0.15708)) - y_local * sin(count * (0.15708)) - x_local;
121 y_offset_rotate = x_local * sin(count * (0.15708)) + y_local * cos(count * (0.15708)) - y_local;
122 }
123 else if(side == 0 && count >= 11) {
124 current_yaw = current_yaw - (count - 11) * (0.15708);
125 x_offset_rotate = x_local * cos(-(count - 11) * (0.15708)) - y_local * sin(-(count - 11) * (0.15708)) - x_local;
126 y_offset_rotate = x_local * sin(-(count - 11) * (0.15708)) + y_local * cos(-(count - 11) * (0.15708)) - y_local;
127 }
128 else if(side == 1 && count < 11) {
129 current_yaw = current_yaw - count * (0.15708);
130 x_offset_rotate = x_local * cos(-count * (0.15708)) - y_local * sin(-count * (0.15708)) - x_local;
131 y_offset_rotate = x_local * sin(-count * (0.15708)) + y_local * cos(-count * (0.15708)) - y_local;
132 }
133 else if(side == 1 && count >= 11) {
134 current_yaw = current_yaw + (count - 11) * (0.15708);
135 x_offset_rotate = x_local * cos((count - 11) * (0.15708)) - y_local * sin((count - 11) * (0.15708)) - x_local;
136 y_offset_rotate = x_local * sin((count - 11) * (0.15708)) + y_local * cos((count - 11) * (0.15708)) - y_local;
137 }
138 tf2::Quaternion q1;
139 q1.setRPY(current_roll, current_pitch, current_yaw);
140 // end_effector pose
141 TomoPose eef_pose;
142 eef_pose.pos[0] = current_eef_pose.pos[0] + offset_pose_1.pos[0] - x_offset_rotate;
143 eef_pose.pos[1] = current_eef_pose.pos[1] + offset_pose_1.pos[1] - y_offset_rotate;
144 eef_pose.pos[2] = current_eef_pose.pos[2] + offset_pose_1.pos[2];
145 eef_pose.quat = q1;
146 return eef_pose;
147 }
148 std::vector<double> calculateCorrectedHeadJoints(const std::vector<double>& joints, const std::vector<std::vector<double>>& coef,
149 const std::vector<std::vector<double>>& power, const std::vector<double>& intercept)

151 if(coef.size() == 0 || power.size() == 0 || intercept.size() == 0) {
